\setkomafont{pagenumber}{\usekomafont{pagenumberOriginal}}%
\addtokomafont{pagenumber}{\color{DenKrKomaColor_PageNumber}}%
%
% \clearpairofpagestyles%
% Remove the Page-Number from the Pagestyle-Pair  [plain.scrheadings]{scrheadings}
% NOTE: In one-sided printing, only right-hand pages exist, and LATEX designates these as odd pages regardless of their page number.
\lefoot[]{}%
\rofoot[]{}%
\cofoot[]{}% Clearing the central pagenumber of one-sided
%Define Layer for new Page-Number Style
\providecommand*\pagenumberruleA{}\renewcommand*\pagenumberruleA{%
  {\color{DenKrKomaColor_ChapterHeading}\rule[-90pt]{1.25pt}{100pt}}%
}%
\providecommand*\pagenumberruleB{}\renewcommand*\pagenumberruleB{%
{\color{DenKrKomaColor_ChapterHeading}\rule[-90pt]{0.625pt}{100pt}}%
}%
\DeclareLayer[%
  background,%
  oddpage,%
  foot,%
  contents={%
    \hfill%
    \makebox[2.125pt][l]{%
      \pagenumberruleB\hspace*{1.5pt}%
      \pagenumberruleA\hspace*{10pt}%
      {\pagemark}%
    }%
  }%
]{pagenumber.odd}%
\DeclareLayer[%
  background,%
  evenpage,%
  foot,%
  contents={%
    \makebox[2.125pt][r]{%
      {\pagemark}%
      \hspace*{10pt}\pagenumberruleA%
      \hspace*{1.5pt}\pagenumberruleB%
    }%
  }%
]{pagenumber.even}%
\AddLayersToPageStyle{scrheadings}{pagenumber.odd,pagenumber.even}%
\AddLayersToPageStyle{plain.scrheadings}{pagenumber.odd,pagenumber.even}%
% % % % %
\DenKrKOMALayerTrackAdd{scrheadings}{pagenumber.odd,pagenumber.even}%
\DenKrKOMALayerTrackAdd{plain.scrheadings}{pagenumber.odd,pagenumber.even}%
