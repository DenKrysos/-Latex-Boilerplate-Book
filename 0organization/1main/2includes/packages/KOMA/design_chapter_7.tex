\def\chapSplit#1#2{%
    \StrSplit{#2}{#1}\tempa\tempb%
    \contour{DenKrKomaColor_ChapterHeading_Outline}{\color{DenKrKomaColor_ChapterHeading}\tempa}%
    \let\tempa\empty%
    \unless\ifx\tempb\empty\def\tempa{\\\chapSplit{#1}{\tempb}}\fi%
    \tempa%
}%
