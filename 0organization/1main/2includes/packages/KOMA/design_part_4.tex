\setkomafont{part}{\sffamily\bfseries\huge\color{DenKrKomaColor_PartHeading}}%
\renewcommand*{\partformat}{%
    {\usekomafont{part}\partname~\thepart}%
}%
%
\makeatletter%
\renewcommand*\partlineswithprefixformat[3]{%
    \Ifstr{#1}{part}{%
        \begin{center}%
        \begin{tikzpicture}[remember picture,baseline=(num.base)]%
            \ifstrempty{#2}{%
                \node[draw=none,shape=rectangle,anchor=base,inner sep=0ex](num){};%
            }{%
                \node[draw=none,shape=rectangle,anchor=base,inner sep=0ex](num){\partformat};%
            }%
            \node[draw=none,shape=rectangle,anchor=north,inner sep=0ex](name)at($(num.south)+(0,-2cm)$){#3};%
            \coordinate(crossC)at($(num.south)!.5!(name.north)$);%
            \path let\p1=(0.15cm,0.8cm),\p2=(4cm,0.15cm)in%
                coordinate(c1T)at($(crossC)+(0,\y1)$)%
                coordinate(c1TL)at($(crossC)+(-\x1,0.3*\y1)$)%
                coordinate(c1TR)at($(crossC)+(\x1,0.3*\y1)$)%
                coordinate(c1B)at($(crossC)+(0,-\y1)$)%
                coordinate(c1BL)at($(crossC)+(-\x1,-0.4*\y1)$)%
                coordinate(c1BR)at($(crossC)+(\x1,-0.4*\y1)$)%
                %
                coordinate(c2L)at($(crossC)+(-\x2,0)$)%
                coordinate(c2LT)at($(crossC)+(-0.5*\x2,\y2)$)%
                coordinate(c2LB)at($(crossC)+(-0.5*\x2,-\y2)$)%
                coordinate(c2R)at($(crossC)+(\x2,0)$)%
                coordinate(c2RT)at($(crossC)+(0.5*\x2,\y2)$)%
                coordinate(c2RB)at($(crossC)+(0.5*\x2,-\y2)$)%
            ;%
            \path[draw=none,fill=DenKrKomaColor_SectionHeading!70!gray,line width=0pt,line cap=round,line join=round]%
                let\n1={40},\n2={0}, \n3={20},\n4={15}in%
            (crossC)%
                to[out=90+\n1,in=270+\n2]%
                (c1TL)%
                to[in=270-\n1,out=90-\n2]%
                (c1T)%
                to[out=270+\n1,in=90+\n2]%
                (c1TR)%
                to[in=90-\n1,out=270-\n2]%
                cycle
            (crossC)%
                to[out=270-\n1,in=90-\n2]%
                (c1BL)%
                to[in=90+\n1,out=270+\n2]%
                (c1B)%
                to[out=90-\n1,in=270-\n2]%
                (c1BR)%
                to[in=270+\n1,out=90+\n2]%
                cycle
            (crossC)%
                to[out=180-\n3,in=0-\n4]%
                (c2LT)%
                to[in=0+\n3,out=180+\n4]%
                (c2L)%
                to[out=0-\n3,in=180-\n4]%
                (c2LB)%
                to[in=180+\n3,out=0+\n4]%
                cycle
            (crossC)%
                to[out=0+\n3,in=180+\n4]%
                (c2RT)%
                to[in=180-\n3,out=0-\n4]%
                (c2R)%
                to[out=180+\n3,in=0+\n4]%
                (c2RB)%
                to[in=0-\n3,out=180-\n4]%
                cycle
            ;%
        \end{tikzpicture}%
        \end{center}%
    }{%
        \@hangfrom{#2}{#3}% other section levels using style=part
    }%
}%
\makeatother%
