    \coordinate(ShadeCtrl2To)at(ShadeOrigin);%

    \ChapShadeCorner%

    %
    \coordinate(ShadeFrom1)at(chapter_title.south \nodeAnchOpposite);%
    \coordinate(ShadeFrom2)at($(chapter_line_B)+(\ShiftDirectionOpp 1pt,0)$);%
    \coordinate(ShadeOrigin)at(ShadeFrom1-|ShadeFrom2);%

    \ChapShadeCorner%
