\def\chapSplit#1#2{%
    \StrSplit{#2}{#1}\tempa\tempb%
    \tempa\let\tempa\empty%
    \unless\ifx\tempb\empty\def\tempa{\\\chapSplit{#1}{\tempb}}\fi%
    \tempa%
}%
